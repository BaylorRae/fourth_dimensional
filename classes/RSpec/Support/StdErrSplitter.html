def method_missing(name, *args, &block)

def write(line)
  return if line =~ %r{^\S+/gems/\S+:\d+: warning:} # http://rubular.com/r/kqeUIZOfPG

  @orig_stderr.write(line)
  @output_tracker.write(line)
end
