def test_path_map_works_on_pathnames
  pathname = Pathname.new("this/is/a/dir/abc.rb")
  assert_equal Pathname.new("abc.rb"), pathname.pathmap("%f")
  assert_equal Pathname.new("this/is/a/dir"), pathname.pathmap("%d")
end
