def add(path, url)
  path = path.split(File::SEPARATOR) unless Array === path
  @children ||= {}
  if path.length == 1
    @children[path.first] = url
  else
    @children[path.first] ||= FilesTree.new
    @children[path.first].add(path[1, path.length], url)
  end
end
