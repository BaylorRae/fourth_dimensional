def test_near_system_dir_isnt_suppressed
  path = RbConfig::CONFIG['rubylibprefix']
  skip if path.nil?
  path = File.expand_path path

  paths = [" " + path + ":12"]

  actual = Rake::Backtrace.collapse(paths)

  assert_equal paths, actual
end
