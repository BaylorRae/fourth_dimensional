def test_with_99_exit_status
  s = Rake::PseudoStatus.new(99)
  assert_equal 99, s.exitstatus
  assert_equal 25344, s.to_i
  assert_equal 99, s >> 8
  refute s.stopped?
  assert s.exited?
end
