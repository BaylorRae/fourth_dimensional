def test_no_deprecated_dsl
  rakefile %q{
    task :check_task do
      Module.new { p defined?(task) }
    end

    task :check_file do
      Module.new { p defined?(file) }
    end
  }

  assert_equal "nil", invoke_normal(:check_task).chomp
  assert_equal "nil", invoke_normal(:check_file).chomp
end
